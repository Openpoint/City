/**
 * Check if a value is in an array
 * @param {string | bolean | object | array} value - needle
 * @param {array} array - haystack
 * @returns {bolean}
 * @extends CORE
 * */
function isInArray(value, array) {
  return array.indexOf(value) > -1;
}
